JavaScript Code for Decimal to Binary Conversion

2023-07-31

JavaScript program that converts decimal to binary.

        
          function toBinary (decNumber) {
            let bitString = '';
            let number = decNumber;
          
            while (number > 0) {
                bitString += (number % 2).toString();
                number = Math.floor(number / 2);
            }
          
            return bitString
              .split('')
              .reverse()
              .join('')
          }
          
          console.log(toBinary(2));
          console.log(toBinary(4));
          console.log(toBinary(6));
          console.log(toBinary(44));
          console.log(toBinary(100));

            10
            100
            110
